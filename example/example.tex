\begin{theorem}\label{thm:murphys}
If something is suspicious...
\begin{proof}
Let $n$ be odd. Then $n = 2k+1$.  
Observe that
\[
( k+1 )^2 - ( k )^2 = (k^2 + 2k + 1) - k^2 = 2k+1 = n.
\]
Thus every odd number is a difference of two squares.
\end{proof}
\end{theorem}
